#include <iostream>
#include <stdlib.h>
#include <time.h>

using namespace std;

int min(int A[], int n, int i, int m); // функция нахождения наименьшего элемента в массиве
// A - массив
// n - количество элементов в массиве
// i - номер элемента, с которого нужно начать поиск
// m - текущий минимум (то, с чем сравниваем)

int main()
{
    int A[20]; // Создаем массив на 10 элементов
    srand(time(0)); // запускаем генератор случайных чисел
    for(int i=0; i<20; i++) // каждому элементу массива от 0-го по (N-1)-й
        A[i] = rand()%201 - 100; // присваиваем случайное значение от -100 до 100
    
    for(int i=0; i<20; i++) // выводим каждый элемент массива на экран
        cout << A[i] << " ";
    cout << endl;
    
    cout << "Наименьший отрицательный элемент равен " << min(A, 20, 0, 0);
    
    return 0;
}

int min(int A[], int n, int i, int m){
    int mm; // новый текущий минимум
    if(A[i]<m) mm = A[i]; // обновляем, если элемент массива оказался еще меньше
    else mm = m; // если нет оставляем старое значение
    if(i==n-1) return mm; // если i==n-1 значит дошли до конца массива и должны дать ответ
    else return(min(A, n, i+1, mm)); // иначе проверяем следующий элемент
}
