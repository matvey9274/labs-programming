#include <iostream>

using namespace std;

int minDiv(int N, int d); // функция нахождения минимального делителя N, начиная с d

int main()
{
    int N;
    cout << "Введите число, которое нужно проверить: ";
    cin >> N;
    
    // проверяем наименьший делитель числа N, больший чем 1
    if(minDiv(N,2)==N) // если наименьший делитель числа N равен N, значит число простое
        cout << "Это простое число";
    else // иначе N делится на что-то между 1 и N, значит оно составное
        cout << "Это составное число";

    return 0;
}

int minDiv(int N, int d){ // функция нахождения минимального делителя N, начиная с d
    if(N % d == 0) return d; // если N делится на d, говорим, что ответ d
    else return minDiv(N, d+1); // иначе запускаем функцию заново и проверяем d+1
}
