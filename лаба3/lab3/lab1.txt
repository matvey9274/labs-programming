#include <iostream>
#include <string>

using namespace std;

string copy(string s, int n); // функция, копирующая первые n символов строки s в возвращаемую строку

int main()
{
    setlocale( LC_ALL,"Russian" );
    string s1, s2;
    int n;
    cout << "Введите строку, из которой нужно скопировать символы:" << endl;
    cin >> s1;
    cout << "Сколько символов скопировать?" << endl;
    cin >> n;
    s2 = copy(s1, n); // используем функцию для копирования первых n сиволов из строки s1 в строку s2
    cout << "Новая строка из " << n << " первых символов: " << endl;
    cout << s2; // выводим результат
}

string copy(string s, int n){ // функция, копирующая первые n символов строки s в возвращаемую строку
    string k = "";  // сначала строка пустая
    int l = s.length(); // длина строки, из которой происходит копирование
    int i = 0; // обнуляем счетчик
    while((i<n) && (i<l)){ // пока счетчик не досчитал до конца копируемой строки и до n
        k += s[i]; // копируем очередной символ в строку k
        i++; // увеличиваем счетчик
    }
    return k; // возвращаем полученную строку k в качестве результата
}
