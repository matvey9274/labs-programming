#include <iostream>
#include <fstream>
#include <string>

using namespace std;

int main()
{
    string words[1000]; // массив для считывания слов из файла
    string s; // переменная для временного хранения слова
    
	ifstream f("text.txt"); // открываем файл text.txt для чтения
    int cnt=0; // счетчик слов изначально равен 0
    while(f >> s){ // пока в файле есть слова сохраняем очередное слово в переменной s
        words[cnt] = s; // записываем считанное слово в массив слов
        cnt++; // увеличиваем счетчик слов
    }
	f.close(); // закрываем файл text.txt
    
    for(int i=0; i<cnt; i++){ // сортируем массив слов по убыванию длин (сортировка пузырьком)
        for(int j=cnt-1; j>i; j--){ // проходим по неотсортированной части массива
            if(words[j].length() > words[j-1].length()){ // если предыдущее слово короче текущего
                string t = words[j]; // меняем 
                words[j] = words[j-1]; // их
                words[j-1] = t;			// местами
            }
        }
    }
    
	// вывод слов в нужном порядке
    for(int i=0; i<cnt; i++) // проходим по отсортированному массиву слов
        cout << words[i] << endl; // выводим очередное слово

    return 0;
}
