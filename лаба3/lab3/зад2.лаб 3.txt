#include <iostream>
#include <fstream>
#include <string>

using namespace std;

int main()
{
    string text[100]; // массив, в который текст из файла будет сохранен построчно
    string words[100]; // массив, в который будут записаны слова из последней строки
    int line = 0; // счетчик для строк, считываемых из файла
    string s; // переменная, в которую считывается очередная строка

	ifstream f("text.txt"); // открываем файл text.txt для чтения
    while(getline(f,s)){ // пока есть очередная строка сохраняем ее в переменную s
        text[line] = s; // считанную строку записываем в очередную ячейку массива s
        line++; // увеличиваем номер очередной ячейки
    }
	f.close(); // закрываем файл text.txt
    string lastline = text[line-1]; // в переменную lastline помещаем последнюю считанную строку

	// теперь нужно разделить эту строку на слова
    int j=0; // счетчик слов изначально равен 0
    for(int i=0; i<lastline.length(); i++){ // проходим по всем символам строки
        if(lastline[i]!=' ') // если символ - не пробел 
            words[j] += lastline[i]; // дописываем его к последнему слову в массиве words
        else // иначе (если встретился пробел)
            j++; // увеличиваем счетчик слов
    }
    
    string word; // переменная для самого длинного слова
    int maxl = 0; // изначально наибольшая длина слова равна 0
    
    for(int i=0; i<j; i++){ // проходим по всем словам в массиве words
        if(words[i].length() > maxl){ // если длина очередного слова больше, чем наибольшая длина
            maxl = words[i].length(); // перезаписываем наибольшую длину
            word = words[i]; // самое длинное на данный момент слово сохраняем в переменной word
        }
    }
    // после просмотра всех слов самое длинное останется в переменной word
    cout << word; // выводим

    return 0;
}
